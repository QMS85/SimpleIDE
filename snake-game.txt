    <span id="score">Score: 0</span>
  </div>

<!-- JS -->
<script>
const canvas = document.getElementById('snake-canvas');
const ctx = canvas.getContext('2d');
const SIZE = 20, COUNT = canvas.width / SIZE;
let snake, dx, dy, food, score = 0, interval, gameOver = false;

function resetGame() {
  snake = [{x:5, y:5}];
  dx = 1; dy = 0;
  score = 0; gameOver = false;
  document.getElementById('score').textContent = "Score: " + score;
  spawnFood();
}

function draw() {
  ctx.fillStyle = "#222";
  ctx.fillRect(0,0,canvas.width,canvas.height);
  ctx.fillStyle = "#60a5fa";
  snake.forEach(seg => ctx.fillRect(seg.x*SIZE, seg.y*SIZE, SIZE-2, SIZE-2));
  ctx.fillStyle = "#eab308";
  ctx.fillRect(food.x*SIZE, food.y*SIZE, SIZE-2, SIZE-2);
}

function spawnFood() {
  food = {
    x: Math.floor(Math.random()*COUNT),
    y: Math.floor(Math.random()*COUNT)
  };
  if(snake.some(seg=>seg.x===food.x && seg.y===food.y)) spawnFood();
}

function tick() {
  if(gameOver) return;
  const head = {x: snake[0].x + dx, y: snake[0].y + dy};
  if(head.x<0 || head.x>=COUNT || head.y<0 || head.y>=COUNT ||
     snake.some(seg=>seg.x===head.x && seg.y===head.y)) {
    gameOver = true; clearInterval(interval);
    alert('Game Over! Your score: '+score);
    return;
  }
  snake.unshift(head);
  if(head.x === food.x && head.y === food.y) {
    score += 10;
    document.getElementById('score').textContent = "Score: " + score;
    spawnFood();
  } else {
    snake.pop();
  }
  draw();
}

document.addEventListener('keydown', e => {
  if(gameOver) return;
  if(e.key==='ArrowLeft' && dx!==1) { dx=-1; dy=0; }
  if(e.key==='ArrowRight' && dx!==-1) { dx=1; dy=0; }
  if(e.key==='ArrowUp' && dy!==1) { dx=0; dy=-1; }
  if(e.key==='ArrowDown' && dy!==-1) { dx=0; dy=1; }
});

document.getElementById('startBtn').onclick = function() {
  resetGame();
  draw();
  clearInterval(interval);
  interval = setInterval(tick, 120);
};
</script>
